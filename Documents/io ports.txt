0x00-0x07	Timer (8254)
0x00		channel 0 data port
0x02		channel 1 data port
0x04		channel 2 data port
0x06		command register
--------------------------------------------------------------------------------
0x08-0x0F	PIC (8259)
0x08		command port
0x0A		data port
--------------------------------------------------------------------------------
0x10-0x17	PPI (8255)
0x10		Port A
0x12		Port B
0x14		Port C
0x16		Control Port
--------------------------------------------------------------------------------
0x20-0x27	Keyboard controller (8042)
--------------------------------------------------------------------------------
0x27-0x2F	UART (16550)
--------------------------------------------------------------------------------
0x28-0x2F	VGA
0x28		VGA control port
0x2A		background color

control port:	--ihvpmm

mm	r/w	graphics mode (0 = text, 1 = 320x200x8)
p	r/w	active plane				
v	r	vertical retrace (1 = active)
h	r	horizontal retrace (1 = active)
i	r/w	enable vertical retrace interrupt
--------------------------------------------------------------------------------
